def resumen_eda(df):
    print("📊 Dimensiones del dataset")
    print(f"Filas: {df.shape[0]}")
    print(f"Columnas: {df.shape[1]}")
    print("-" * 40)

    print("\n🔎 Tipos de datos")
    print(df.dtypes)
    print("-" * 40)

    print("\n🔎 Primeros registros")
    print(df.head(2))
    print("-" * 40)

    print("\n📈 Estadísticas descriptivas")
    print(df.describe(include='number'))
    print("-" * 40)

    print("\n📈 Estadísticas descriptivas")
    try:
        print(df.describe(include='object', exclude='str'))
    except:
      print("No hay variables categóricas")
    print("-" * 40)

    print("\n⚠️ Valores nulos por columna")
    nulos = df.isnull().sum()
    if (nulos > 0).any():
        print(nulos[nulos > 0].sort_values(ascending=False))
    else:
        print("No hay valores nulos en el DataFrame (0 en todas las columnas)")
    print("-" * 40)

    print("\n📌 Porcentaje de nulos por columna")
    porcentaje_nulos = df.isnull().mean() * 100
    if (porcentaje_nulos > 0).any():
        print(porcentaje_nulos[porcentaje_nulos > 0].sort_values(ascending=False))
    else:
        print("No hay valores nulos (0% en todas las columnas)")
    print("-" * 40)

    print("\n🔁 Registros duplicados")
    duplicados = df.duplicated().sum()
    print(f"Cantidad de duplicados: {duplicados}")
    print("-" * 40)

Acciones en ventas:

import pandas as pd

# Crear copia y renombrar columna
df_ven_pp = df_ven.copy()
df_ven_pp.rename(columns={'precio': 'precio_unitario'}, inplace=True)

# Limpieza de precio: quitar '$' y convertir a float
df_ven_pp['precio_unitario'] = df_ven_pp['precio_unitario'].str.replace('$', '', regex=False)
df_ven_pp['precio_unitario'] = pd.to_numeric(df_ven_pp['precio_unitario'], errors='coerce')

# Eliminar duplicados
df_ven_pp.drop_duplicates(inplace=True)

# Eliminar nulos (vimos que hay 2 en precio y 2 en cantidad)
df_ven_pp.dropna(subset=['precio_unitario', 'cantidad'], inplace=True)

# Asegurar que cantidad sea entero
df_ven_pp['cantidad'] = df_ven_pp['cantidad'].astype(int)

print("✅ Limpieza completada.")

    (df_roi_final['fecha_venta'] <= df_roi_final['fecha_fin'])
)

# 2. Agrupación final por canal
roi_final_canal = df_roi_final[df_roi_final['venta_en_campanha']].groupby('canal').agg(
    ingresos_atribuidos=('venta_neta', 'sum'),
    costo_total_marketing=('costo', 'sum'),
    cantidad_ventas=('id_venta', 'count')
)

# 3. Cálculo de ROI final
roi_final_canal['ROI'] = roi_final_canal['ingresos_atribuidos'] / roi_final_canal['costo_total_marketing']

print("📈 Resumen de Performance por Canal (Fechas Corregidas):")
display(roi_final_canal.sort_values(by='ROI', ascending=False).style.format({

    'costo_total_marketing': '${:,.2f}',
    'ROI': '{:.2f}x'
}))
